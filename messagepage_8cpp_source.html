00061 void MessagePage::on_sendButton_clicked()
00062 {
00063     QString text = ui->messageTextEdit->toPlainText().trimmed();
00064     if (text.isEmpty() || !m_contact->isOnline())
00065         return;
00066 
00067     QccPacket message;
00068     message.stream() << qint32(qrand()) << m_contact->username() << m_contact->encrypt(text);
00069     message.send(m_socket);
00070 
00071     text.replace('\n', "<br/>");
00072     ui->messagesTextEdit->append("<span style=\"color: #00f;\">You</span>: " + text);
00073 
00074     ui->messageTextEdit->clear();
00075     ui->messageTextEdit->setFocus();
00076 }

00083 bool MessagePage::eventFilter(QObject *obj, QEvent *event)
00084 {
00085     if (obj == ui->messageTextEdit) {
00086         if (event->type() == QEvent::KeyPress) {
00087             QKeyEvent *keyEvent = static_cast<QKeyEvent*>(event);
00088             if ((keyEvent->key() == Qt::Key_Return || keyEvent->key() == Qt::Key_Enter)
00089                     && keyEvent->modifiers() == Qt::NoModifier) {
00090                 on_sendButton_clicked();
00091                 return true;
00092             } else
00093                 return false;
00094         } else
00095             return false;
00096     } else
00097         return QWidget::eventFilter(obj, event);
